#### Instructions ####
all: $(SOURCES) $(EXECUTABLE)

$(EXECUTABLE): $(OBJECTS)
    $(CC) $(OBJECTS) -o $@;

.cpp.o:
    $(CC) $(CFLAGS) $< -o $@;

.PHONY: all clean clear run valgrind c new n r v nr nv ar av

# prevents failures resulting from errors returned by rm
clean:
    rm *.o $(EXECUTABLE); clear;

run:
    ./$(EXECUTABLE) $(ARGUMENTS);

# require install
valgrind:
    valgrind $(VFLAGS) ./$(EXECUTABLE) $(ARGUMENTS);



#### Compounds and Shorthands ####
c: clean

new: c all

n: new
r: run
v: valgrind

nr: n r
nv: n v

ar: all r
av: all v
